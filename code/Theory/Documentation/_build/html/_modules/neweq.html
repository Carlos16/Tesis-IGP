[docs]def setReqRM(fDict,K_CP,mP):
    Dis = fDict['Discriminant'](K_CP,mP)
    R1 = fDict['R1'](K_CP,mP)
    if Dis>0:
        bR = fDict['bR'](K_CP,mP)
        a = fDict['denR'](K_CP,mP)
        return [R1,(bR+np.sqrt(Dis))/(2*a) , (bR-np.sqrt(Dis))/(2*a)]
    elif Dis==0:
        bR = fDict['bR'](K_CP,mP)
        a =  fDict['denR'](K_CP,mP)
        return  [R1,bR/(2*a)]
    else:
        return [R1]
