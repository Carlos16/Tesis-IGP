[docs]def set_R_C_eq_sRM(m_C,r,K,q1,q1_0,alfa_1,e_1,t_hc,hC_0):
    halfa_1 = alfa_1/m_C
    Req = q1 /(halfa_1*(e_1 - q1_0*hC_0))
    Ceq = (1+t_hc*alfa_1*Req)/halfa_1 *r*(1-Req/K)
